function firstDuplicateCharacter(array){
  var hashSet = {};
  for(var i = 0; i < array.length; i++){
    if(hashSet[array[i]] == undefined){
      hashSet[array[i]] = false;
    } else {
      hashSet[array[i]] = true;
      return array[i]
    }
  }
  return "no duplicate character.";
}

In the code above, we loop though an array and each time we check if the array item exists in the hashSet object. If not, we insert the item into the hashSet. If yes, we return the item value.

function firstNonRepeatingCharacter(string){
  for(var i = 0; i < string.length; i++){
    c1 = string.slice(i, i+1);
    match = false;
    for(var b=0; b < string.length; b++){
    c2 = string.slice(b, b+1);
      if(i != b){
        if(c1 == c2){
          match = true;
          break;
        }
      }
    }
    if(match == false){
      return c1;
    }
  }
}

In the code, we split a character from the string, and then we compare it with each character in the string. If the characters matches, then we break from the loop and compare the character with the second character. If we find no match, then we return the character value.
